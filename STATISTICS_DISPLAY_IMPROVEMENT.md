    st.write(f"**{title}:**")
    
    for channel, values in stats.items():
        st.write(f"**{channel}:**")
        col_mean, col_std, col_min, col_max = st.columns(4)
        
        with col_mean:
            st.metric("Mean", f"{values['Mean']:.2f}")
        with col_std:
            st.metric("Std Dev", f"{values['Std']:.2f}")
        with col_min:
            st.metric("Min", f"{values['Min']}")
        with col_max:
            st.metric("Max", f"{values['Max']}")
